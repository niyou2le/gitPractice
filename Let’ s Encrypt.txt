acme.sh --issue -d raman.htvision.com.cn  -w /usr/share/nginx/html

acme.sh --installcert -d raman.htvision.com.cn \
--keypath       /usr/domains/raman.htvision.com.cn/ssl/firefore.key  \
--fullchainpath /usr/domains/raman.htvision.com.cn/ssl/firefore.pem \
--reloadcmd     "sudo service nginx force-reload"


acme.sh --installcert -d 308.htvision.com.cn \
--keypath       /usr/domains/308.htvision.com.cn/ssl/firefore.key  \
--fullchainpath /usr/domains/308.htvision.com.cn/ssl/firefore.pem \
--reloadcmd     "sudo service nginx force-reload"


listen 443;
	server_name 308.htvision.com.cn;
	root /usr/share/nginx/html;
	index index.html index.htm;


ssl on;
 
        ssl_certificate  /usr/domains/308.htvision.com.cn/ssl/firefore.pem;
        ssl_certificate_key  /usr/domains/308.htvision.com.cn/ssl/firefore.key;
 
        ssl_session_cache shared:SSL:10m;
        ssl_session_timeout 10m;
